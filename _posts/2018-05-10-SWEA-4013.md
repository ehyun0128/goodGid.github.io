## [1] Answer Code (18. 05. 10)

``` cpp

#include <iostream>
#include <cstring>
using namespace std;

int node[4][8];
int visit[4];

void turn(int idx,int dir){
    if(dir==1){
        int tmp = node[idx][7];
        for(int i=7; i>0; i--)
            node[idx][i] = node[idx][i-1];
        node[idx][0] = tmp;
    }
    else{
        int tmp = node[idx][0];
        for(int i=0; i<7; i++)
            node[idx][i] = node[idx][i+1];
        node[idx][7] = tmp;
    }
}

void dfs(int idx, int dir){
    if(visit[idx])
        return ;
    
    visit[idx] = 1;
    if(idx+1<4 && node[idx][2] != node[idx+1][6])
        dfs(idx+1,dir * -1);
    if(idx-1>=0 && node[idx][6] != node[idx-1][2])
        dfs(idx-1,dir*-1);
    
    turn(idx,dir);
    visit[idx] = 0;
    
}

int main(){
    int TC;
    cin >> TC;
    
    for(int tc=1; tc<=TC; tc++){
        memset(node,0,sizeof(node));
        memset(visit,0,sizeof(visit));
        
        int k;
        cin >> k;
        
        for(int i=0; i<4; i++){
            for(int j=0; j<8; j++){
                scanf("%d",&node[i][j]);
            }
        }
        
        for(int i=0; i<k; i++){
            int a,b;
            cin >> a >> b;
            dfs(a-1,b);
        }
        
        int ans = 0;
        if(node[0][0] == 1) ans += 1;
        if(node[1][0] == 1) ans += 2;
        if(node[2][0] == 1) ans += 4;
        if(node[3][0] == 1) ans += 8;
        
        printf("#%d %d\n",tc,ans);
    }
    
    
    return 0;
}
```
