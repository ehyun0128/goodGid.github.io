## [1] Answer Code (18. 09. 21)

``` cpp
#include <iostream>
#include <queue>
using namespace std;

int visit[100001];

int main(){
    int st,dest;
    cin >> st >> dest;
    
    queue<int> q;
    q.push(st);
    
    int time = -1;
    int flag = 0;
    while (! q.empty()) {
        if( flag )
            break;
        
        time ++;
        int size = (int) q.size();
        
        for(int i=0; i<size; i++){
            int top = q.front();
            q.pop();
            
            if( top == dest ){
                flag = 1;
                break;
            }
            if( visit[top] ) // Already Visit
                continue;
            visit[top] = 1;
            
            if( top - 1 >= 0 && !visit[top - 1] )  // Already Visit
                q.push(top - 1);
            
            if(top + 1 <= 100000 && !visit[top + 1] )  // Already Visit
                q.push(top + 1);
            
            if(top * 2 <= 100000 && !visit[top * 2] )  // Already Visit
                q.push(top * 2);
        } // end of for i
    } // end of while
    
    if( flag )
        cout << time << endl;
    return 0;
}
```
