## [1] Answer Code (18. 04. 03)


``` cpp

#include<iostream>
using namespace std;

int funGCM(int a, int b){
    return (a % b == 0 ? b : funGCM(b,a%b));
}

int main(){
    int a,b;
    cin >> a >> b;
    int gcm = funGCM(a, b);
    int lcm = a * b / gcm;
    cout << gcm << endl;
    cout << lcm << endl;
    return 0;
}



```
