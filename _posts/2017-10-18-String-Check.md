* 문자열 비교

{% highlight cpp %}

#include <cstring> // toupper 사용

 bool cmpStr(string s1, string s2) {
    if(s1.length() != s2.length()) {
        return false;
    }
    int len = s1.length();
    for(int i = 0; i < len; i++) {
        if(toupper(s1[i]) != toupper(s2[i]))
            return false;
    }
    return true;
 }

* 띄어 쓰기 단위로 문자열 자르기 (1)

* `char`형 변수 사용 

{% highlight cpp %}

int main(){
    char input[100];
    int size;
    
    cin.getline(input,100);
    
    // 입력받은 문자열 길이 구하기
    int i=0;
    while (1) {
        if(input[i] == '\0'){
            size = i ;
            break;
        }
        i++;
    }
    
    for (int i = 0; i < size; i++) {
        if(input[i] == ' '){
            input[i] = '\n';
        }
    }
    printf("%s\n",input);
    
    return 0;
}

int main(){
    string s;
    getline(cin,s);

    int size = (int)s.size();
    for(int i=0; i<size; i++){
        if( s[i] == ' '){
            cout << " Space " << endl;
        }
        else {
            cout << s[i];
        }
    }
    cout << endl;
    return 0;
}
