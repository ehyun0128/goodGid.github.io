## [1] Answer Code (18. 07. 24)

{% highlight cpp %}

vector<long long> solution(int x, int n) {
    vector<long long> answer;
    for(int i=0; i<n; i++){
        answer.push_back(x + i*x);
    }
    return answer;
}
